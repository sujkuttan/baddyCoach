# Pipeline Accuracy Improvement (Benchmark-Driven, Holistic) — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Raise pipeline output accuracy on the manual-label benchmark (`labels_enriched_new.csv`, 99 labeled shots in the first ~2 min of `test_match.mp4`) across three dimensions — temporal (hit-frame), attribution (who hit), and stroke type — with no regression in any dimension, measured by a reproducible benchmark script.

**Architecture:** Establish a benchmark harness that matches each run's `shots.parquet` to the manual labels (by ground-truth `label_frame`) and reports frame error, stroke exact/similar accuracy, attribution accuracy, and BST-eligibility coverage. Then fix, in leverage order: (1) temporal hit-frame calibration, (2) attribution — wire the alternation prior + full-rally Viterbi into the ownership scorer (the current code drops both), (3) stroke — fix joint center-alignment features and evaluate a checkpoint swap. Every change is re-measured against the labels before the next begins.

**Tech Stack:** Python 3.12, pandas, numpy, pytest. Pipeline modules: `backend/app/pipeline/hits.py`, `backend/app/pipeline/attribution.py`, `backend/app/pipeline/shared/ownership_scorer.py`, `backend/app/pipeline/shared/ownership_quality.py`, `backend/app/pipeline/strokes.py`, `backend/app/models/bst.py`, `backend/app/pipeline/shared/bst_preproc.py`, `backend/app/config/settings.py`.

## Global Constraints

- All tunables live in `backend/app/config/settings.py` (pydantic `Settings`); never hardcode new magic numbers in stage modules.
- Logging uses `PipelineLogger` with **keyword args** (`logger.info("msg", key=val)`), never printf-style positional args.
- Edits follow existing module patterns; do not restructure large files beyond the targeted fix.
- Every task ends with a test (new or existing) that passes and a commit.
- The benchmark is the source of truth: run it after every task and record the four metrics (frame error median, stroke exact+similar %, attribution %, eligibility coverage %). No task regresses any of them vs the prior task's recorded numbers.
- Benchmark ground truth = columns `label_frame` (true frame), `side` (true near/far), `true_stroke`, `true_class_id` from the labels CSV. The `shot_frame`/`frame_diff`/`predicted_*` columns in the CSV are prior-run outputs and are regenerated by the benchmark each run.

```python
# backend/scripts/benchmark_labels.py
import argparse, json, sys
from pathlib import Path
import numpy as np, pandas as pd

SIM = {
 "lift":["lift","defensivelift","clear"],"clear":["clear","lift","defensivelift"],
 "drop":["drop","netshot"],"netshot":["netshot","drop","push"],
 "drive":["drive","push","rush"],"smash":["smash","rush","drive"],
 "serve":["shortserve","longserve","serve"],"push":["push","drive","netshot"],
 "rush":["rush","drive","smash","push"],"block":["block","drive","netshot"],
 "defensivelift":["lift","defensivelift","clear"],
 "softliftorpush":["softliftorpush","lift","push"],
 "crosscourt":["crosscourt","drive","clear"],
 "netshot":["netshot","drop","push"],
 "shortserve":["shortserve","longserve"],"longserve":["longserve","shortserve"],
}
def norm(s): return str(s).strip().lower().replace(" ","").replace("_","")
def stroke_matches(p, l):
    p, l = norm(p), norm(l)
    if p == l: return "exact"
    for k, alts in SIM.items():
        if norm(k) == l and p in [norm(a) for a in alts]: return "similar"
        if norm(k) == p and l in [norm(a) for a in alts]: return "similar"
    return "wrong"

def benchmark(labels: pd.DataFrame, shots: pd.DataFrame, radius: int = 15) -> dict:
    shots = shots.sort_values("frame").reset_index(drop=True)
    unused = set(shots.index)
    matches = []
    for _, lb in labels.iterrows():
        lf = int(lb["label_frame"]); best=None; bd=np.inf
        for idx in unused:
            d = abs(shots.loc[idx, "frame"] - lf)
            if d < bd and d <= radius: bd=d; best=idx
        if best is not None:
            matches.append((best, bd)); unused.discard(best)
        else:
            matches.append((None, None))
    n_matched = sum(1 for b,_ in matches if b is not None)
    fe, sr, pl, elig = [], [], [], []
    for (b, d), (_, lb) in zip(matches, labels.iterrows()):
        if b is None:
            sr.append("missed"); pl.append("missed"); fe.append(np.nan); elig.append(False); continue
        s = shots.loc[b]
        sr.append(stroke_matches(s["stroke_type"], lb["true_stroke"]))
        pl.append("correct" if str(s.get("side","")).strip().lower()==str(lb["side"]).strip().lower() else "wrong")
        fe.append(d)
        elig.append(bool(s.get("bst_input_eligible", False)))
    fe_clean = [f for f in fe if not np.isnan(f)]
    exact = sum(1 for x in sr if x=="exact"); similar = sum(1 for x in sr if x=="similar")
    acc = exact + similar
    n_elig = sum(1 for e in elig if e)
    acc_elig = sum(1 for e,a in zip(elig, sr) if e and a in ("exact","similar"))
    return {
        "n_labels": len(labels), "n_pipeline": len(shots), "n_matched": n_matched,
        "recall": 100*n_matched/max(1,len(labels)),
        "stroke_accuracy_strict": 100*exact/max(1,n_matched),
        "stroke_accuracy": 100*acc/max(1,n_matched),
        "player_accuracy": 100*sum(1 for p in pl if p=="correct")/max(1,n_matched),
        "mean_frame_error": float(np.nanmean(fe_clean)) if fe_clean else 0.0,
        "median_frame_error": float(np.nanmedian(fe_clean)) if fe_clean else 0.0,
        "max_frame_error": float(np.nanmax(fe_clean)) if fe_clean else 0.0,
        "coverage": 100*n_elig/max(1,n_matched),
        "accepted_accuracy": 100*acc_elig/max(1,n_elig) if n_elig else 0.0,
    }

def main():
    ap = argparse.ArgumentParser()
    ap.add_argument("--labels", default="labels_enriched_new.csv")
    ap.add_argument("--shots", default="results/hybrid_results/debug/shots.parquet")
    ap.add_argument("--radius", type=int, default=15)
    ap.add_argument("--quiet", action="store_true")
    a = ap.parse_args()
    labels = pd.read_csv(a.labels); labels = labels[labels["label_status"]=="labeled"] if "label_status" in labels else labels
    shots = pd.read_parquet(a.shots)
    m = benchmark(labels, shots, radius=a.radius)
    if not a.quiet:
        print("=== BENCHMARK (labels vs this run's shots) ===")
        for k,v in m.items(): print(f"  {k}: {v:.2f}" if isinstance(v,float) else f"  {k}: {v}")
    else:
        print(json.dumps(m))

if __name__ == "__main__":
    main()
```
